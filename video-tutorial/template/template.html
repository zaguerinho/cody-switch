Chapter 2 Creating the Express Server
1 // server.js - Express REST API
2 const express = require('express');
3 const cors = require('cors');
4
5 const app = express();
6 const PORT = process.env.PORT || 3000;
7
8 // Middleware
9 app.use(cors());
10 app.use(express.json());
11
12 // In-memory data store
13 const todos = [
14 { id: 1, title: 'Learn Express', completed: true },
15 { id: 2, title: 'Build REST API', completed: false },
16 ];
17
18 // GET all todos
19 app.get('/api/todos', (req, res) => {
20 res.json(todos);
21 });
22
23 // GET single todo
24 app.get('/api/todos/:id', (req, res) => {
25 const todo = todos.find(t => t.id === parseInt(req.params.id));
26 if (!todo) {
27 return res.status(404).json({ error: 'Todo not found' });
28 }
29 res.json(todo);
30 });
31
32 // POST create todo
33 app.post('/api/todos', (req, res) => {
34 const { title } = req.body;
35 if (!title) {
36 return res.status(400).json({ error: 'Title is required' });
37 }
38 const newTodo = {
39 id: todos.length + 1,
40 title,
41 completed: false
42 };
43 todos.push(newTodo);
44 res.status(201).json(newTodo);
45 });
46
47 // Start server
48 app.listen(PORT, () => {
49 console.log(`Server running on port ${PORT}`);
50 });
The GET all todos endpoint is the simplest route handler — it returns the full array as JSON with a 200 status.
